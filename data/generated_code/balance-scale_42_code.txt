
# (Left-Weighted Distance Difference)
# Usefulness: This feature captures the difference in weighted distances between left and right sides, which can help the model learn the importance of each side in the balance scale.
# Input samples: 'left-weight': [3.0, 5.0, 3.0], 'left-distance': [5.0, 5.0, 4.0], 'right-distance': [2.0, 3.0, 2.0]
df['left_weighted_distance'] = df['left-weight'] * df['left-distance']

# (Right-Weighted Distance Difference)
# Usefulness: This feature captures the difference in weighted distances between right and left sides, which can help the model learn the importance of each side in the balance scale.
# Input samples: 'right-weight': [5.0, 3.0, 5.0], 'right-distance': [2.0, 3.0, 2.0], 'left-distance': [5.0, 5.0, 4.0]
df['right_weighted_distance'] = df['right-weight'] * df['right-distance']
